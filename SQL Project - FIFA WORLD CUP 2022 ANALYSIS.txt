#44. WRITE a query to show average avg_age of each group and arrange it in descending order by avg_age.
SELECT 
    grp_stats.grp_no, 
    team_data.avg_age,
    AVG(team_data.avg_age) 
FROM team_data
INNER JOIN grp_stats
    ON team_data.team = grp_stats.team
GROUP BY grp_stats.grp_no,team_data.avg_age
ORDER BY team_data.avg_age DESC;
